	$(document).ready(function(){
		$('#cancellTask').hide();
		
		$('#showForm').on('click',function () {
			$("#addTask").load('/task/add-form');
			$("#addTask").show();
			$("#showForm").hide();
			$('#cancellTask').show();
			
		});
		$('#cancellTask').on('click',function(){

			$('#cancellTask').hide();
			$('#showForm').show();
		});
